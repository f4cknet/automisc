**`src/automisc/tools/misc/archive/qemu_img.py`** (新, ~100 LOC):

```python
"""qemu_img adapter — 虚拟磁盘探测 (info 命令, per 铁律 7 探测归探测).

实战 vmdk/qcow2/raw 等格式 (per v0.5-train-018 flag.vmdk).
输出 SP: vdisk_format (虚拟磁盘格式识别) + file_header (顶层 magic 命中).

可挂 auto-run (探测归探测), per 铁律 7.
"""
from automisc.core.registry import register_tool
from automisc.core.result import ToolResult
from automisc.core.suspicious import SuspiciousPoint
from automisc.tools.base import ToolAdapter


@register_tool
class QemuImgAdapter(ToolAdapter):
    name = "qemu_img"
    category = "archive"
    description = "qemu-img info 虚拟磁盘探测 (vmdk/qcow2/raw 等格式识别, 探测归探测)"

    default_timeout = 30.0  # info 命令快

    def run(self, file_path: str) -> ToolResult:
        from automisc.tools.paths import resolve_tool_binary
        qemu_img_bin = self.binary_path or resolve_tool_binary("qemu-img") or "qemu-img"
        cmd = [qemu_img_bin, "info", file_path]
        exit_code, stdout, stderr, duration_ms = self._run_subprocess(cmd, timeout=self.default_timeout)

        suspicious: list[SuspiciousPoint] = []

        # 1. 通用扫描 (owner 铁律: 关键字命中 = 可疑)
        from automisc.tools.suspicious_scanner import scan_output_for_suspicious
        suspicious.extend(scan_output_for_suspicious(
            tool_name=self.name, file_path=file_path, stdout=stdout,
        ))

        # 2. 虚拟磁盘格式识别
        combined_lower = (stdout + "\n" + stderr).lower()
        if "file format:" in combined_lower or "virtual size:" in combined_lower:
            for line in stdout.split("\n"):
                line_strip = line.strip()
                if line_strip.lower().startswith("file format:"):
                    fmt = line_strip.split(":", 1)[1].strip()
                    suspicious.append(SuspiciousPoint(
                        id="",
                        tool_name=self.name,
                        file_path=file_path,
                        category="vdisk_format",
                        offset=None,
                        matched_pattern=f"qemu-img 识别虚拟磁盘格式: {fmt}",
                        severity=3,  # info 级别 (跟 exiftool 同)
                        suggested_action=(
                            f"虚拟磁盘 {fmt}, 走 GUI 工具栏 '🖼️ qemu-img 转换' 转 raw 后 "
                            f"foremost 雕内嵌文件 (per v0.5-train-018 实战路径)"
                        ),
                    ))
                elif line_strip.lower().startswith("virtual size:"):
                    size_str = line_strip.split(":", 1)[1].strip()
                    # 累计到 file_format SP 即可, 不重复
                    pass

        return ToolResult(
            tool_name=self.name,
            exit_code=exit_code,
            stdout=stdout,
            stderr=stderr,
            suspicious_points=suspicious,
        )
```

**`src/automisc/tools/misc/archive/qemu_img_extract.py`** (新, ~80 LOC):

```python
"""qemu_img_extract adapter — 虚拟磁盘转换 (convert 命令, 写盘操作).

实战 vmdk/qcow2 转换 raw (per v0.5-train-018 flag.vmdk).
GUI 工具栏 '🖼️ qemu-img 转换' 入口, 不挂 auto-run (per 铁律 7 写盘).

输出: <input_stem>__qemu_img_raw/<input_stem>.raw (per v0.5-output-samedir).
"""
from pathlib import Path
import shutil
from automisc.core.registry import register_tool
from automisc.core.result import ToolResult
from automisc.core.suspicious import SuspiciousPoint
from automisc.tools.base import ToolAdapter
from automisc.core.utils.output_path import extract_dir_for


@register_tool
class QemuImgExtractAdapter(ToolAdapter):
    name = "qemu_img_extract"
    category = "archive"
    description = "qemu-img convert 虚拟磁盘转换 (vmdk/qcow2 → raw, 写盘, GUI 工具栏手工)"

    default_timeout = 120.0  # 大文件转换可能慢

    def run(self, file_path: str) -> ToolResult:
        from automisc.tools.paths import resolve_tool_binary
        qemu_img_bin = self.binary_path or resolve_tool_binary("qemu-img") or "qemu-img"

        # 1. 准备 output 目录 (per v0.5-output-samedir)
        extract_dir = extract_dir_for(file_path, purpose="qemu_img_raw")
        if extract_dir.exists():
            shutil.rmtree(extract_dir)
        extract_dir.mkdir(parents=True, exist_ok=True)
        output_raw = extract_dir / f"{Path(file_path).stem}.raw"

        # 2. 构造 qemu-img convert 命令 (per v0.5-train-018 实战)
        #    默认 -f vmdk -O raw, 实战 flag.vmdk 走 vmdk → raw
        #    进阶: 留 -f/-O 透传 (v0.5+ 实战覆盖 qcow2/raw 时加)
        cmd = [
            qemu_img_bin,
            "convert",
            "-f", "vmdk",     # TODO: 自动探测 (info 一次)
            "-O", "raw",
            file_path,
            str(output_raw),
        ]
        exit_code, stdout, stderr, duration_ms = self._run_subprocess(cmd, timeout=self.default_timeout)

        suspicious: list[SuspiciousPoint] = []
        combined = (stdout + "\n" + stderr).lower()

        # 1. 通用扫描
        from automisc.tools.suspicious_scanner import scan_output_for_suspicious
        suspicious.extend(scan_output_for_suspicious(
            tool_name=self.name, file_path=file_path, stdout=stdout,
        ))

        # 2. 成功信号: 退出码 0 + output raw 存在
        if exit_code == 0 and output_raw.exists():
            size = output_raw.stat().st_size
            suspicious.append(SuspiciousPoint(
                id="",
                tool_name=self.name,
                file_path=file_path,
                category="vdisk_extracted",
                offset=None,
                matched_pattern=f"qemu-img 转换 vmdk → raw, {size} bytes → {output_raw}",
                severity=5,  # 关键成功信号 (per v0.5-sevenz-extract 模式)
                suggested_action=(
                    f"qemu-img 转换成功, output: {output_raw} — "
                    f"用 foremost 雕 (per v0.5-train-018 实战路径)"
                ),
            ))

        return ToolResult(
            tool_name=self.name,
            exit_code=exit_code,
            stdout=stdout,
            stderr=stderr,
            suspicious_points=suspicious,
            metadata={"written_files": [{"path": str(output_raw), "kind": "raw vdisk"}]} if output_raw.exists() else [],
                      "extract_dir": str(extract_dir)},
        )
```

## 5. 风险

| 风险 | 概率 | 缓解 |
|---|---|---|
| qemu-img adapter 写盘操作污染 auto-run pool (race condition 类似) | 低 | qemu_img (info 探测) **可**挂 auto-run, qemu_img_extract (convert 写盘) **不**挂; v0.5-fix-find-suspicious-race-condition 已修 base.py Popen + _terminate_current_proc 强 kill, 写盘工具中途 kill 不会留 .raw 半成品 (shutil.rmtree 重写) |
| 实战 1 道同类升架构过早 (§5.2 标准做法) | 中 | owner 23:23 拍 B 破例, 实战需求强 (装完期望 GUI 入口); 后续实战 ≥ 1 道同类时再 review spec 决策 |
| qemu-img convert 大文件 (10GB+) 超时 | 中 | default_timeout 120s 实战够, 大文件走 GUI 工具栏手工触发 (per 七z extract 模式), 不阻塞 auto-run |
| qemu-img 装失败 (PATH 无) | 低 | adapter 用 `resolve_tool_binary("qemu-img")` 找, 找不到 fall back to "qemu-img" → subprocess FileNotFoundError → exit 127, SP 报告 (per v0.5-sevenz-extract 失败兜底) |
| v0.5-zipcrack-doc-update 同模式 "✅ 已装 但**没** GUI 入口" | 中 | 本 spec **直接**建 GUI 入口 (不写 doc 弥补), owner memory 教训"不要凭空造新风格" — 严格按 v0.5-sevenz-extract 模式 (2 adapter + 双注册 + menu_dock) 实施 |
| qemu_img_extract convert 命令硬编码 -f vmdk -O raw | 高 | 实战 1 道 (flag.vmdk) vmdk 固定; 未来 qcow2/raw 实战时加 -f/-O 透传 (per spec §6 Q1) |
